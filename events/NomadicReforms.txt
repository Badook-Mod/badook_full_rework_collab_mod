			}

				add_province_triggered_modifier = treaty_port_western

			684 = {
				add_province_triggered_modifier = treaty_port_western
			}
			
			#battle sites

			
			5088 = {
				add_province_triggered_modifier = battle_of_yamen

				add_province_triggered_modifier = battle_of_fei_river
