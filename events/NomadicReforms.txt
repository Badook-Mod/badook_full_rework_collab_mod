country_event = {
	id = nomadic_events.1
	title = nomadic_events.1.t
	desc = nomadic_events.1.d
	picture = MESOAMERICAN_GODS_eventPicture
	
	trigger = {
		is_in_war = {
			attacker_leader = ROOT
			war_score = 50
			casus_belli = cb_unity_gain
		}
		OR = {    	#for cheese uncons
			unit_in_battle = yes
			war_score = 99
		}
		NOT = {
			has_country_flag = unity_gained 
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = unity_gained
		}
	}
	option = {
		name = nomadic_events.1.a #keep me in
		add_horde_unity = 25
	}
}
