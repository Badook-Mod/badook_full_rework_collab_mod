# limited war bonus
country_event = {
	id = boutyboutique.5
	title = "boutyboutique.5t"
	desc = "boutyboutique.5d"
	picture = CITY_DEVELOPMENT_eventPicture
	hidden = yes
	
	trigger = {
		NOT = { has_country_modifier = limited_war }
		OR = {
			is_in_war = {
				casus_belli = cb_limited_war
				attacker_leader = ROOT
			}
			is_in_war = {
				casus_belli = cb_limited_imperial_war
				attacker_leader = ROOT
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "boutyboutique.05"
		add_country_modifier = { 
			name = limited_war
			duration = 1830 
		}
	}
}

	id = boutyboutique.21
	title = "EVTNAME725"
	desc = "EVTDESC725"
	picture = CITY_DEVELOPMENT_eventPicture
	fire_only_once = yes
	
	trigger = {
		is_year = 1443
		ai = yes
	}

	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = "boutyboutique.O2"
		every_province = {
			limit = {
				OWNER = { 
					NOT = { has_institution = feudalism } 
				}
			}
			add_institution_embracement = {
				which = feudalism
				value = 100
			}
		}
		every_country = {
			limit = {
				NOT = { adm_tech = 3 }
			}
			add_adm_tech = 1
		}
		every_country = {
			limit = {
				NOT = { dip_tech = 3 }
			}
			add_dip_tech = 1
		}
		every_country = {
			limit = {
				NOT = { mil_tech = 3 }
			}
			add_mil_tech = 1
		}
		every_country = {
			limit = {
				religion_group = muslim
			}
			country_event = { id = muslim_school_events.20 days = 1 }
		}
	}
}
